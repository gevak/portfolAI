import numpy as np
import matplotlib.pyplot as plt
from collections import defaultdict

# Import the detection algorithm from glglstats module
from glglstats.detection import find_repetitions
from glglstats.visualization import plot_repetition_matrix

class SongRepetitionAnalyzer:
    """
    Analyzes and visualizes repetition patterns in songs.
    
    GlglStats is a novel algorithm for detecting repeated sections
    in songs, even with tempo changes and small variations.
    """
    
    def __init__(self, audio_path):
        self.audio_path = audio_path
        self.segments = []
        self.repetition_matrix = None
        
    def analyze(self):
        # Load and segment the audio file
        audio_data = self.load_audio(self.audio_path)
        
        # Find repetitions using the core algorithm
        self.repetition_matrix, self.segments = find_repetitions(
            audio_data, 
            window_size=4096,
            similarity_threshold=0.85
        )
        
        return self.repetition_matrix

    def visualize(self):
        # Generate the repetition plot using the matrix
        figure = plot_repetition_matrix(
            self.repetition_matrix,
            title=f"Repetition Analysis: {self.audio_path.split('/')[-1]}",
            highlight_patterns=True
        )
        return figure
