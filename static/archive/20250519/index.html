def calculate_compression_ratio(input_data):
    # Implement LZ77 compression
    from io import BytesIO
    import zlib
    
    compressed = zlib.compress(input_data.encode())
    ratio = 1.0 - len(compressed) / len(input_data)
    return ratio * 100.0  # Return as percentage

// C++ implementation of polygon orientation algorithm
void orientPolygon(Polygon& poly, const Vector2D& direction) {
    // Find extreme vertices
    auto extremePoints = findExtremeVertices(poly);
    
    // Calculate stable orientation
    double minPotential = std::numeric_limits<::double>::max();
    double stableAngle = 0.0;
    
    for (double angle = 0; angle < 2 * M_PI; angle += 0.01) {
        double potential = calculateGravitationalPotential(poly, angle, direction);
        if (potential < minPotential) {
            minPotential = potential;
            stableAngle = angle;
        }
    }
    
    poly.rotate(stableAngle);
}

function typeWriter(element, text, speed = 30, callback = null) {
    let i = 0;
    element.innerHTML = '';
    
    function type() {
        if (i < text.length) {
            element.innerHTML += text.charAt(i);
            i++;
            setTimeout(type, speed + Math.random() * 50);
        } else if (callback) {
            setTimeout(callback, 500);
        }
    }
    
    type();
}
